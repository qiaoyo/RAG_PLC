```st
VAR_INPUT
	UTC : UDINT;
	DST_ENABLE : BOOL;
	TIME_ZONE_OFFSET : INT;
END_VAR

VAR
	tmp : INT;
END_VAR

tmp := TIME_ZONE_OFFSET * 60 + BOOL_TO_INT(DST_ENABLE AND DST(UTC)) * 3600;
IF tmp < 0 THEN
	UTC_TO_LTIME := UTC - _INT_TO_UDINT(ABS(tmp));
ELSE
	UTC_TO_LTIME := UTC + _INT_TO_UDINT(tmp);
END_IF;
```
