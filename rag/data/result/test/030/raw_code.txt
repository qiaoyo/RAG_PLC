```st
VAR_INPUT
STR : STRING;
END_VAR

VAR_OUTPUT
IS_NUM : BOOL;
END_VAR

VAR
pos : INT;
I : INT;
END_VAR

(*@KEY@: WORKSHEET
NAME: IS_NUM
IEC_LANGUAGE: ST
*)
I := LEN(str);
FOR pos := 1 TO I DO
IF NOT ISC_NUM(INT_TO_BYTE(GET_CHAR(str,pos))) THEN
IS_NUM := FALSE;
RETURN;
END_IF;
END_FOR;
IS_NUM := I > 0;
```
