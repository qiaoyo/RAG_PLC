```st
VAR_INPUT
    I1 : BOOL;
    I2 : BOOL;
    TL : TIME;
END_VAR

VAR_OUTPUT
    Q1 : BOOL;
    Q2 : BOOL;
END_VAR

VAR
    T1 : TOF;
    T2 : TOF;
END_VAR

(*@KEY@: WORKSHEET
NAME: INTERLOCK
IEC_LANGUAGE: ST
*)
(* 输入信号有一个运行延迟，用于锁定另一个输入 *)
T1(IN := I1, PT := TL);
T2(IN := I2, PT := TL);

Q1 := I1 AND NOT T2.Q;
Q2 := I2 AND NOT T1.Q;
```
